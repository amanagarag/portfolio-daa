Case Study 1: Unavailability of School Bus

Problem

Students have to travel long distances on foot, which can be dangerous, especially during extreme weather conditions.

Impact

High absenteeism, students arriving late, and physical exhaustion.

Solution

1. Dijkstra's Algorithm for Optimal Route Selection

#include <iostream>
#include <vector>
#include <queue>
#include <climits>
using namespace std;

struct Edge {
    int dest, weight;
};

void dijkstra(int start, vector<vector<Edge>>> &graph, vector<int> &distances) {
    priority_queue<pair<int, int>>, vector<pair<int, int>>, greater<> pq;
    distances[start] = 0;
    pq.push({0, start});

    while (!pq.empty()) {
        auto [dist, node] = pq.top();
        pq.pop();

        for (const auto& edge : graph[node]) {
            int newDist = dist + edge.weight;
            if (newDist < distances[edge.dest]) {
                distances[edge.dest] = newDist;
                pq.push({newDist, edge.dest});
            }
        }
    }
}

int main() {
    int nodes = 5;
    vector<vector<Edge>>> graph(nodes);

    graph[0].push_back({1, 10});
    graph[0].push_back({2, 3});
    graph[1].push_back({3, 2});
    graph[2].push_back({1, 1});
    graph[2].push_back({3, 8});

    vector<int> distances(nodes, INT_MAX);
    dijkstra(0, graph, distances);

    for (int i = 0; i < nodes; i++)
        cout << "Distance to node " << i << ": " << distances[i] << endl;
    return 0;
}

2. Impact of Improved Transportation

Impact: Reduced absenteeism and safer travel for students.

Case Study 2: No Ambulance Putting Life at Risk

Problem

Critical medical emergencies are delayed due to no ambulance availability.

Impact

Increased mortality rates and overcrowded local healthcare facilities.

Solution

1. Hungarian Algorithm for Ambulance Allocation

#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

int calculateDistance(pair<int, int> a, pair<int, int> b) {
    return abs(a.first - b.first) + abs(a.second - b.second);
}

int main() {
    vector<pair<int, int>> hospitals = {{0, 0}, {2, 2}};
    vector<pair<int, int>> emergencies = {{1, 1}, {3, 3}};

    for (const auto& hospital : hospitals) {
        for (const auto& emergency : emergencies) {
            cout << "Distance from hospital (" << hospital.first << ", " << hospital.second << ") to emergency (" 
                 << emergency.first << ", " << emergency.second << ") is " << calculateDistance(hospital, emergency) << endl;
        }
    }
    return 0;
}

2. Impact of Ambulance Routing

Impact: Quicker response times and saved lives.

Case Study 3: Unavailability of Solar System

Problem

Lack of infrastructure to install and maintain solar systems in remote areas.

Impact

Dependency on non-renewable energy sources and higher energy costs.

Solution

1. Genetic Algorithm for Solar Panel Placement

#include <iostream>
#include <vector>
#include <cstdlib>
using namespace std;

struct Solution {
    vector<int> placement;
    int fitness;
};

void evaluate(Solution& sol) {
    sol.fitness = 0;
    for (int x : sol.placement) {
        sol.fitness += x * x;
    }
}

int main() {
    Solution initialSolution = {{1, 0, 1, 0, 1}, 0};
    evaluate(initialSolution);
    cout << "Fitness of initial solution: " << initialSolution.fitness << endl;
    return 0;
}

2. Impact of Solar System Availability

Impact: Enhanced access to renewable energy and reduced energy costs.
